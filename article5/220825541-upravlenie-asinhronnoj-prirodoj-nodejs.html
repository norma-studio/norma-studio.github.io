Управление асинхронной природой Node.js...

CodeWeb Development

Интро

Node.js позволяет легко и быстро создавать приложения. Но из-за его асинхронного характера языка может оказаться трудным написать читаемый и управляемый код. В этой статье я покажу вам несколько советов о том, как добиться этого....

Node.js позволяет легко и быстро создавать приложения. Но из-за его асинхронного характера языка может оказаться трудным написать читаемый и управляемый код. В этой статье я покажу вам несколько советов о том, как добиться этого.

Ад обратных вызовов или пирамида судьбы

Node.js построен таким образом, чтобы вы могли использовать асинхронные функции. Это означает обратные вызовы, обратные вызовы и еще больше обратных вызовов. Вы, наверное, видели или даже пишете подобные фрагменты кода:

app.get("/login", function (req, res) {  sql.query("SELECT 1 FROM users WHERE name = ?;", [ req.param("username") ], function (error, rows) {if (error) {res.writeHead(500);return res.end();}if (rows.length &lt; 1) {res.end("Wrong username!");} else {sql.query("SELECT 1 FROM users WHERE name = ? && password = MD5(?);", [ req.param("username"), req.param("password") ], function (error, rows) {if (error) {res.writeHead(500);return res.end();}if (rows.length &lt; 1) {res.end("Wrong password!");} else {sql.query("SELECT * FROM userdata WHERE name = ?;", [ req.param("username") ], function (error, rows) {if (error) {res.writeHead(500);return res.end();}req.session.username = req.param("username");req.session.data = rows[0];res.rediect("/userarea");});}});}});});

На самом деле это фрагмент взят прямо из одного из моих первых приложений Node.js. Если вы сделали что-то более продвинутое в Node.js, вы, вероятно, все понимаете, но проблема здесь в том, что код перемещается вправо каждый раз, когда вы используете какую-либо асинхронную функцию. Сложнее читать и сложнее отлаживать. К счастью, для этого беспорядка есть несколько решений, поэтому вы можете выбрать подходящее для своего проекта.

Решение 1. Именование и модульность обратных вызовов

Самый простой подход - назвать каждый обратный вызов (который поможет вам отлаживать код) и разделить весь ваш код на модули. Приведенный выше пример входа в систему можно превратить в модуль несколькими простыми шагами.

Структура

Начнем с простой структуры модуля. Чтобы избежать вышеупомянутой ситуации, когда вы просто разделите беспорядок на более мелкие беспорядки, давайте сделаем его классом:

var util = require("util");function Login(username, password) {function _checkForErrors(error, rows, reason) {}function _checkUsername(error, rows) {}function _checkPassword(error, rows) {}function _getData(error, rows) {}function perform() {}this.perform = perform;}util.inherits(Login, EventEmitter);

Класс создан с двумя параметрами: username и password. Если посмотреть пример кода, нам понадобятся три функции: одна, чтобы проверить правильность имени пользователя (_checkUsername), другая - проверить пароль (_checkPassword) и еще одна, чтобы вернуть связанные с пользователем данные (_getData) и уведомить приложение о том, что логин был успешным. Существует также хелпер _checkForErrors, который будет обрабатывать все ошибки. Наконец, есть функция perform, которая запустит процедуру входа (и является единственной публичной функцией в классе). Наконец, мы наследуемся от EventEmitter, чтобы упростить использование этого класса.

Помощник

Функция _checkForErrors проверяет, произошла ли какая-либо ошибка или если SQL-запрос не возвращает строки, и испускает соответствующую ошибку (по причине, которая была поставлена):

function _checkForErrors(error, rows, reason) {if (error) {this.emit("error", error);return true;}if (rows.length &lt; 1) {this.emit("failure", reason);return true;}return false;}

Она также возвращает true или false, в зависимости от того, произошла ли ошибка или нет.

Выполнение входа в систему

Функция perform должна выполнить только одну операцию: выполнить первый запрос SQL (чтобы проверить, существует ли имя пользователя) и назначить соответствующий обратный вызов:

function perform() {sql.query("SELECT 1 FROM users WHERE name = ?;", [ username ], _checkUsername);}

Я предполагаю, что ваше SQL-соединение доступно глобально в переменной sql (просто для упрощения, обсуждение, является ли это хорошей практикой, выходит за рамки этой статьи). Функция готова.

Проверка имени пользователя

Следующий шаг - проверить правильность имени пользователя и, если это так, второй запрос - проверить пароль:

function _checkUsername(error, rows) {if (_checkForErrors(error, rows, "username")) {return false;} else {sql.query("SELECT 1 FROM users WHERE name = ? && password = MD5(?);", [ username, password ], _checkPassword);}}

Совсем тот же код, что и в беспорядочном образце, за исключением обработки ошибок.

Проверка пароля

Эта функция почти такая же, как и предыдущая, единственная разница заключается в том, что запрос вызывается:

function _checkPassword(error, rows) {if (_checkForErrors(error, rows, "password")) {return false;} else {sql.query("SELECT * FROM userdata WHERE name = ?;", [ username ], _getData);}}

Получение данных, связанных с пользователем

Последняя функция в этом классе получит данные, связанные с пользователем (необязательный шаг), и запустит с ним событие успеха:

function _getData(error, rows) {if (_checkForErrors(error, rows)) {return false;} else {this.emit("success", rows[0]);}}

Заключительные штрихи и использование

Последнее, что нужно сделать, это экспортировать класс. Добавьте эту строку после всего кода:

module.exports = Login;

Это сделает класс Login единственным, что модуль будет экспортировать наружу. Его можно впоследствии использовать так (предполагая, что вы назвали файл модуля login.js, и он находится в том же каталоге, что и основной скрипт):

var Login = require("./login.js");...app.get("/login", function (req, res) {var login = new Login(req.param("username"), req.param("password));login.on("error", function (error) {res.writeHead(500);res.end();});login.on("failure", function (reason) {if (reason == "username") {res.end("Wrong username!");} else if (reason == "password") {res.end("Wrong password!");}});login.on("success", function (data) {req.session.username = req.param("username");req.session.data = data;res.redirect("/userarea");});login.perform();});

Вот еще несколько строк кода, но читаемость кода значительно увеличилась. Кроме того, это решение не использует никаких внешних библиотек, что делает его идеальным, если кто-то новый приходит в ваш проект.

Это был первый подход, давайте перейдем ко второму.

Решение 2: Обещания

Использование обещаний - еще один способ решения этой проблемы. Обещание (как вы можете прочитать в предоставленной ссылке) «представляет собой возможное значение, возвращаемое из одного завершения операции». На практике это означает, что вы можете связать вызовы, чтобы сгладить пирамиду и сделать код более удобным для чтения.

Мы будем использовать модуль Q, доступный в репозитории NPM.

Q в двух словах

Прежде чем мы начнем, позвольте мне познакомить вас с Q. Для статических классов (модулей) мы будем в первую очередь использовать функцию Q.nfcall. Это помогает нам в преобразовании каждой функции, следующей за Node.js обратным вызовом (где параметры обратного вызова являются ошибкой и результатом) к обещанию. Это используется следующим образом:

Q.nfcall(http.get, options);

Это похоже на Object.prototype.call. Вы также можете использовать Q.nfapply, который напоминает Object.prototype.apply:

Q.nfapply(fs.readFile, [ "filename.txt", "utf-8" ]);

Кроме того, когда мы создаем обещание, мы добавляем каждый шаг с помощью метода then(stepCallback), улавливаем ошибки с помощью catch(errorCallback) и заканчиваем с done().

В этом случае, поскольку объект sql является экземпляром, а не статическим классом, мы должны использовать Q.ninvoke или Q.npost, которые аналогичны приведенным выше. Разница заключается в том, что мы передаем имя метода как строку в первом аргументе и экземпляр класса, с которым мы хотим работать как второй, чтобы избежать того, что метод не привязан к экземпляру.

Подготовка обещания

Первое, что нужно сделать, это выполнить первый шаг, используя Q.nfcall или Q.nfapply (используйте тот, который вам больше нравится, нет никакой разницы):

var Q = require("q");...app.get("/login", function (req, res) {Q.ninvoke("query", sql, "SELECT 1 FROM users WHERE name = ?;", [ req.param("username") ])});

Обратите внимание на отсутствие точки с запятой в конце строки - вызовы функций будут скованы так, чтобы их не было. Мы просто вызываем sql.query, как в беспорядочном примере, но мы опускаем параметр обратного вызова - он обрабатывается обещанием.

Проверка имени пользователя

Теперь мы можем создать обратный вызов для SQL-запроса, он будет почти идентичен примеру в примере «пирамида судьбы». Добавьте это после вызова Q.ninvoke:

.then(function (rows) {if (rows.length &lt; 1) {res.end("Wrong username!");} else {return Q.ninvoke("query", sql, "SELECT 1 FROM users WHERE name = ? && password = MD5(?);", [ req.param("username"), req.param("password") ]);}})

Как видите, мы добавляем обратный вызов (следующий шаг) с использованием метода then. Кроме того, в обратном вызове мы опускаем параметр error, потому что позже мы поймаем все ошибки. Мы проверяем вручную, если запрос возвращает что-либо, и если так, мы возвращаем следующее обещание, которое будет выполнено (опять же, точка с запятой из-за цепочки).

Проверка пароля

Как и в случае с модуляцией, проверка пароля почти идентична проверке имени пользователя. Это должно произойти сразу после последнего вызова then:

.then(function (rows) {if (rows.length &lt; 1) {res.end("Wrong password!");} else {return Q.ninvoke("query", sql, "SELECT * FROM userdata WHERE name = ?;", [ req.param("username") ]);}})

Получение данных, связанных с пользователем

Последним шагом будет тот, где мы помещаем данные пользователей в сеанс. Еще раз, обратный вызов не сильно отличается от грязного примера:

.then(function (rows) {req.session.username = req.param("username");req.session.data = rows[0];res.rediect("/userarea");})

Проверка ошибок

При использовании обещаний и библиотеки Q все ошибки обрабатываются набором обратных вызовов с использованием метода catch. Здесь мы только отправляем HTTP 500 независимо от того, какая была ошибка, как в приведенных выше примерах:

.catch(function (error) {res.writeHead(500);res.end();}).done();

После этого мы должны вызвать метод done, чтобы «убедиться, что если ошибка не будет обработана до конца, она будет восстановлена и отправлена» (из README библиотеки). Теперь наш красиво сплющенный код должен выглядеть так (и вести себя так же, как и грязный):

var Q = require("q");...app.get("/login", function (req, res) {Q.ninvoke("query", sql, "SELECT 1 FROM users WHERE name = ?;", [ req.param("username") ]).then(function (rows) {if (rows.length &lt; 1) {res.end("Wrong username!");} else {return Q.ninvoke("query", sql, "SELECT 1 FROM users WHERE name = ? && password = MD5(?);", [ req.param("username"), req.param("password") ]);}}).then(function (rows) {if (rows.length &lt; 1) {res.end("Wrong password!");} else {return Q.ninvoke("query", sql, "SELECT * FROM userdata WHERE name = ?;", [ req.param("username") ]);}}).then(function (rows) {req.session.username = req.param("username");req.session.data = rows[0];res.rediect("/userarea");}).catch(function (error) {res.writeHead(500);res.end();}).done();});

Код намного чище, и он требует меньше переписывания, чем подход с модулями.

Решение 3: Step Library

Это решение похоже на предыдущее, но оно проще. Q немного тяжелый, потому что он реализует всю идею обещаний. Библиотека Step существует только с целью сглаживания кучи обратных вызовов. Это также немного проще в использовании, потому что вы просто вызываете единственную функцию, которая экспортируется из модуля, передаете все ваши обратные вызовы в качестве параметров и используйте this вместо каждого обратного вызова. Поэтому беспорядочный пример можно преобразовать в это, используя модуль Step:

var step = require("step");...app.get("/login", function (req, res) {step(function start() {sql.query("SELECT 1 FROM users WHERE name = ?;", [ req.param("username") ], this);},function checkUsername(error, rows) {if (error) {res.writeHead(500);return res.end();}if (rows.length &lt; 1) {res.end("Wrong username!");} else {sql.query("SELECT 1 FROM users WHERE name = ? && password = MD5(?);", [ req.param("username"), req.param("password") ], this);}},function checkPassword(error, rows) {if (error) {res.writeHead(500);return res.end();}if (rows.length &lt; 1) {res.end("Wrong password!");} else {sql.query("SELECT * FROM userdata WHERE name = ?;", [ req.param("username") ], this);}},function (error, rows) {if (error) {res.writeHead(500);return res.end();}req.session.username = req.param("username");req.session.data = rows[0];res.rediect("/userarea");});});

Недостатком здесь является отсутствие общего обработчика ошибок. Хотя любые исключения, вызванные одним обратным вызовом, передаются следующему в качестве первого параметра (поэтому сценарий не будет завершен из-за неперехваченного исключения), причем один обработчик для всех ошибок часто бывает более удобным.

Какой из них выбрать?

Это в значительной степени личный выбор, но чтобы помочь вам выбрать правильный вариант, вот список плюсов и минусов каждого подхода:

Модульность:

Плюсы:

  • Нет внешних библиотек
  • Помогает сделать код более многоразовым

Минусы:

  • Больше кода
  • Много переписывания, если вы конвертируете существующий проект

Обещания (Q):

Плюсы:

  • Меньше кода
  • Только небольшое переписывание, если оно применяется к существующему проекту

Минусы:

  • Вы должны использовать внешнюю библиотеку
  • Требуется немного обучения

Step Library:

Плюсы:

  • Прост в использовании, не требуется обучение
  • Довольно много копировать и вставлять, если конвертировать существующий проект

Минусы:

  • Нет общего обработчика ошибок
  • Немного сложнее сделать step, чтобы эта функция функционировала должным образом

Вывод

Как вы можете видеть, асинхронным характер Node.js можно управлять, и можно избежать кучи обратных вызовов. Я лично использую модульный подход, потому что мне нравится хорошо структурировать мой код. Надеюсь, эти советы помогут вам сделать ваш код более читабельным и отладить ваши сценарии.
